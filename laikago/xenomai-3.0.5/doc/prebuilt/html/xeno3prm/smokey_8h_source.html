136 #define __T(__ret, __action) \
137  ({ \
138  (__ret) = (__action); \
139  if (__ret) { \
140  if ((__ret) > 0) \
141  (__ret) = -(__ret); \
142  smokey_warning("FAILED: %s (=%s)", \
143  __stringify(__action), \

153  smokey_warning("FAILED: %s (=0)", \
154  __stringify(__action)); \
