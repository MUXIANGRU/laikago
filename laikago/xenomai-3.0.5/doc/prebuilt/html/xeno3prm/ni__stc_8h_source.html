315 static inline unsigned int RTSI_Output_Bit(unsigned channel, int is_mseries)
316 {
317  unsigned max_channel;
318  unsigned base_bit_shift;
319  if(is_mseries)
320  {
321  base_bit_shift = 8;
322  max_channel = 7;
323  }else
324  {
325  base_bit_shift = 9;
326  max_channel = 6;
327  }
328  if(channel > max_channel)
329  {
330  rtdm_printk("%s: bug, invalid RTSI_channel=%i\n",
331  __FUNCTION__, channel);
332  return 0;
333  }
334  return 1 << (base_bit_shift + channel);
335 }
