333  const unsigned d_in,
334  unsigned *m_out,
335  unsigned *s_out)
336 {
337  /*
338  * Avoid clever compiler optimizations to occur when d is
339  * known at compile-time. The performance of this function is
340  * not critical since it is only called at init time.
